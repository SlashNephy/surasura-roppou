> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Tech Stack:** React 19, TanStack Router, TypeScript 6, Tailwind CSS 4, Vitest, Testing Library, lucide-react, shadcn/ui style primitives。

Create `src/shared/utils/readability.ts`:

```ts
export type ReadabilityTransformMode =
  "article-number" | "date" | "law-number" | "parentheses" | "unchanged" | "all";

const kanjiDigitValues = new Map([
  ["一", 1],
  ["二", 2],
  ["三", 3],
  ["四", 4],
  ["五", 5],
  ["六", 6],
  ["七", 7],
  ["八", 8],
  ["九", 9],
]);

const kanjiUnitValues = new Map([
  ["千", 1000],
  ["百", 100],
  ["十", 10],
]);

const kanjiNumberPattern = "[一二三四五六七八九十百千]+";

export const toArabicNumber = (kanjiNumber: string): number | undefined => {
  if (kanjiNumber === "" || /[^一二三四五六七八九十百千]/.test(kanjiNumber)) {
    return undefined;
  }

  let total = 0;
  let pendingDigit: number | undefined;

  for (const char of kanjiNumber) {
    const digit = kanjiDigitValues.get(char);
    if (digit !== undefined) {
      if (pendingDigit !== undefined) {
        return undefined;
      }
      pendingDigit = digit;
      continue;
    }

    const unit = kanjiUnitValues.get(char);
    if (unit === undefined) {
      return undefined;
    }

    total += (pendingDigit ?? 1) * unit;
    pendingDigit = undefined;
  }

  return total + (pendingDigit ?? 0);
};

export const transformReadableText = (
  text: string,
  mode: ReadabilityTransformMode = "all",
): string => {
  switch (mode) {
    case "article-number":
      return transformArticleNumbers(text);
    case "date":
      return transformDates(text);
    case "law-number":
      return transformLawNumbers(text);
    case "parentheses":
      return transformParentheses(text);
    case "unchanged":
      return text;
    case "all":
      return transformArticleNumbers(
        transformDates(transformLawNumbers(transformParentheses(text))),
      );
  }
};

const transformParentheses = (text: string): string =>
  text.replaceAll("（", "(").replaceAll("）", ")");

const transformArticleNumbers = (text: string): string => {
  const articleNumberText = text.replace(
    new RegExp(`第(${kanjiNumberPattern})(条|項|号)(?:の(${kanjiNumberPattern}))?`, "g"),
    (match, number, suffix, branchNumber) => {
      const arabicNumber = toArabicNumber(number);
      const arabicBranchNumber =
        branchNumber === undefined ? undefined : toArabicNumber(branchNumber);
      if (
        arabicNumber === undefined ||
        (branchNumber !== undefined && arabicBranchNumber === undefined)
      ) {
        return match;
      }

      return `第${arabicNumber}${suffix}${arabicBranchNumber === undefined ? "" : `の${arabicBranchNumber}`}`;
    },
  );

  return articleNumberText.replace(
    new RegExp(`別表(${kanjiNumberPattern})`, "g"),
    (match, number) => {
      const arabicNumber = toArabicNumber(number);
      return arabicNumber === undefined ? match : `別表${arabicNumber}`;
    },
  );
};

const transformDates = (text: string): string =>
  text.replace(
    new RegExp(
      `(令和|平成|昭和|大正|明治)(${kanjiNumberPattern})年(${kanjiNumberPattern})月(${kanjiNumberPattern})日`,
      "g",
    ),
    (match, era, year, month, day) => {
      const yearNumber = toArabicNumber(year);
      const monthNumber = toArabicNumber(month);
      const dayNumber = toArabicNumber(day);
      if (yearNumber === undefined || monthNumber === undefined || dayNumber === undefined) {
        return match;
      }

      return `${era}${yearNumber}年${monthNumber}月${dayNumber}日`;
    },
  );

const transformLawNumbers = (text: string): string =>
  text.replace(
    new RegExp(
      `(令和|平成|昭和|大正|明治)(${kanjiNumberPattern})年法律第(${kanjiNumberPattern})号`,
      "g",
    ),
    (match, era, year, lawNumber) => {
      const yearNumber = toArabicNumber(year);
      const lawNumberValue = toArabicNumber(lawNumber);
      if (yearNumber === undefined || lawNumberValue === undefined) {
        return match;
      }

      return `${era}${yearNumber}年法律第${lawNumberValue}号`;
    },
  );
```

Note for implementer: if ESLint flags line length in the nested `"all"` expression or RegExp construction, split the expressions across local variables without changing behavior.

const getDisplayText = (node: LawNode, displayMode: LawTextDisplayMode): string => {
  const text = displayMode === "original" ? node.rawText || node.plainText : node.plainText;
  return displayMode === "readable" ? transformReadableText(text) : text;
};

const getDisplayInlineText = (
  text: string | undefined,
  displayMode: LawTextDisplayMode,
): string | undefined => {
  if (text === undefined) {

  return displayMode === "readable" ? transformReadableText(text) : text;
};

Update `stripTrailingChildPlainTexts` signature:

```ts
const stripTrailingChildPlainTexts = (
  plainText: string,
  children: LawNode[],
  displayMode: LawTextDisplayMode,
): string =>
  children.reduceRight((bodyText, child) => {
    const childText = getDisplayText(child, displayMode);
    if (childText === "") {
      return bodyText;
    }

    if (!bodyText.endsWith(childText)) {
      return bodyText;
    }

    return bodyText.slice(0, -childText.length).trim();
  }, plainText);
```
