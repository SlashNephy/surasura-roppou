# ローカルデータ移行 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** export version 2 の全7分類を、プレビュー確認後に既存 IndexedDB へ原子的にマージできる設定画面を実装する。

**Architecture:** `docs/schemas` の draft 2020-12 JSON Schema を Ajv v8 で実行時にも検証し、構文・schema・意味検証を通ったデータだけを `StorageRepository` の単一 transaction へ渡す。Web/Android 共用のテキスト境界は `core/native-integration`、`File`・`Blob`・download は `app` に閉じ込める。

**Tech Stack:** React 19、TypeScript 6、TanStack Router、IndexedDB (`idb`)、Ajv 8.17.1、Vitest、Testing Library、fake-indexeddb、playwright-cli

```ts
import Ajv2020 from "ajv/dist/2020.js";

import savedDataExportSchema from "../../../docs/schemas/saved-data-export-v2.schema.json";

import type { SavedDataExport } from "./export-data";

export type SavedDataImportErrorCode =
  "invalid-json" | "unsupported-version" | "invalid-schema" | "duplicate-id" | "invalid-reference";

export class SavedDataImportError extends Error {
  constructor(
    public readonly code: SavedDataImportErrorCode,
    message: string,
    options?: ErrorOptions,
  ) {
    super(message, options);
    this.name = "SavedDataImportError";
  }
}

export interface SavedDataCounts {
  savedLaws: number;
  bookmarks: number;
  collections: number;
  annotations: number;
  studyCards: number;
  reviewLogs: number;
  studySessions: number;
}

export interface SavedDataImportPreview {
  version: 2;
  exportedAt: string;
  counts: SavedDataCounts;
}

export interface PreparedSavedDataImport {
  data: SavedDataExport;
  preview: SavedDataImportPreview;
}

export interface SavedDataImportResult {
  importedAt: string;
  counts: SavedDataCounts;
}

const ajv = new Ajv2020({ allErrors: false, strict: true });
const validateSavedDataExport = ajv.compile<SavedDataExport>(savedDataExportSchema);

const isRecord = (value: unknown): value is Record<string, unknown> =>
  typeof value === "object" && value !== null && !Array.isArray(value);

const assertUnique = <T>(items: T[], key: (item: T) => string, label: string): void => {
  const ids = new Set<string>();
  for (const item of items) {
    const id = key(item);
    if (ids.has(id)) {
      throw new SavedDataImportError("duplicate-id", `${label} に重複 ID ${id} があります。`);
    }
    ids.add(id);
  }
};

const validateSemantics = (data: SavedDataExport): void => {
  assertUnique(data.savedLaws, (item) => item.law.lawId, "savedLaws");
  assertUnique(data.savedLaws, (item) => item.revision.revisionId, "savedLaws.revision");
  assertUnique(
    data.savedLaws.flatMap((item) => item.nodes),
    (node) => node.id,
    "savedLaws.nodes",
  );
  assertUnique(data.bookmarks, (item) => item.id, "bookmarks");
  assertUnique(data.collections, (item) => item.id, "collections");
  assertUnique(data.annotations, (item) => item.id, "annotations");
  assertUnique(data.studyCards, (item) => item.id, "studyCards");
  assertUnique(data.reviewLogs, (item) => item.id, "reviewLogs");
  assertUnique(data.studySessions, (item) => item.id, "studySessions");

  const cardIds = new Set(data.studyCards.map((card) => card.id));
  for (const log of data.reviewLogs) {
    if (!cardIds.has(log.cardId)) {
      throw new SavedDataImportError(
        "invalid-reference",
        `ReviewLog ${log.id} が存在しない StudyCard ${log.cardId} を参照しています。`,
      );
    }
  }

  for (const savedLaw of data.savedLaws) {
    if (savedLaw.revision.lawId !== savedLaw.law.lawId) {
      throw new SavedDataImportError(
        "invalid-reference",
        `LawRevision ${savedLaw.revision.revisionId} の lawId が一致しません。`,
      );
    }
    for (const node of savedLaw.nodes) {
      if (node.lawId !== savedLaw.law.lawId || node.revisionId !== savedLaw.revision.revisionId) {
        throw new SavedDataImportError(
          "invalid-reference",
          `LawNode ${node.id} の lawId または revisionId が一致しません。`,
        );
      }
    }
  }
};

export const countSavedData = (data: SavedDataExport): SavedDataCounts => ({
  savedLaws: data.savedLaws.length,
  bookmarks: data.bookmarks.length,
  collections: data.collections.length,
  annotations: data.annotations.length,
  studyCards: data.studyCards.length,
  reviewLogs: data.reviewLogs.length,
  studySessions: data.studySessions.length,
});

export const parseSavedDataImport = (input: unknown): PreparedSavedDataImport => {
  if (isRecord(input) && "version" in input && input.version !== 2) {
    throw new SavedDataImportError(
      "unsupported-version",
      `対応していない export version: ${String(input.version)}`,
    );
  }

  if (!validateSavedDataExport(input)) {
    const firstError = validateSavedDataExport.errors?.[0];
    const path = firstError?.instancePath === "" ? "/" : (firstError?.instancePath ?? "/");
    throw new SavedDataImportError(
      "invalid-schema",
      `${path}: ${firstError?.message ?? "JSON Schema に適合しません。"}`,
    );
  }

  validateSemantics(input);

  return {
    data: input,
    preview: {
      version: 2,
      exportedAt: input.exportedAt,
      counts: countSavedData(input),
    },
  };
};
```

```ts
import type { IDBPDatabase } from "idb";

import { fixedIntervalScheduler } from "@/core/study";

import type { SavedDataExport } from "./export-data";
import { countSavedData } from "./import-data";
import type { SavedDataImportResult } from "./import-data";
import type { SurasuraDatabase } from "./schema";
import { withTargetIndexes } from "./stored-record";

const importStoreNames = [
  "laws",
  "lawRevisions",
  "lawNodes",
  "savedLaws",
  "bookmarks",
  "collections",
  "annotations",
  "studyCards",
  "reviewLogs",
  "cardSchedules",
  "studySessions",
] as const;

export const importSavedDataIntoDatabase = async (
  database: IDBPDatabase<SurasuraDatabase>,
  data: SavedDataExport,
  importedAt: string,
): Promise<SavedDataImportResult> => {
  const tx = database.transaction(importStoreNames, "readwrite");

  try {
    const laws = tx.objectStore("laws");
    const lawRevisions = tx.objectStore("lawRevisions");
    const lawNodes = tx.objectStore("lawNodes");
    const savedLaws = tx.objectStore("savedLaws");

    for (const document of data.savedLaws) {
      const existing = await savedLaws.get(document.law.lawId);
      if (existing !== undefined) {
        const nodeKeys = await lawNodes
          .index("by-law-revision")
          .getAllKeys([existing.lawId, existing.revisionId]);
        for (const key of nodeKeys) {
          await lawNodes.delete(key);
        }
        if (existing.revisionId !== document.revision.revisionId) {
          await lawRevisions.delete(existing.revisionId);
        }
      }

      await laws.put(document.law);
      await lawRevisions.put(document.revision);
      for (const [sortOrder, node] of document.nodes.entries()) {
        await lawNodes.put({
          id: node.id,
          lawId: node.lawId,
          revisionId: node.revisionId,
          sortOrder,
          node,
        });
      }
      await savedLaws.put({
        lawId: document.law.lawId,
        revisionId: document.revision.revisionId,
        nodeCount: document.nodes.length,
        savedAt: document.savedAt,
        updatedAt: importedAt,
      });
    }

    const bookmarks = tx.objectStore("bookmarks");
    for (const bookmark of data.bookmarks) {
      await bookmarks.put(withTargetIndexes(bookmark));
    }
    const collections = tx.objectStore("collections");
    for (const collection of data.collections) {
      await collections.put(collection);
    }
    const annotations = tx.objectStore("annotations");
    for (const annotation of data.annotations) {
      await annotations.put(withTargetIndexes(annotation));
    }
    const studyCards = tx.objectStore("studyCards");
    for (const card of data.studyCards) {
      await studyCards.put(withTargetIndexes(card));
    }
    const studySessions = tx.objectStore("studySessions");
    for (const session of data.studySessions) {
      await studySessions.put(session);
    }

    const reviewLogs = tx.objectStore("reviewLogs");
    const affectedCardIds = new Set(data.studyCards.map((card) => card.id));
    for (const log of data.reviewLogs) {
      const previous = await reviewLogs.get(log.id);
      if (previous !== undefined) {
        affectedCardIds.add(previous.cardId);
      }
      affectedCardIds.add(log.cardId);
      await reviewLogs.put(log);
    }

    const cardSchedules = tx.objectStore("cardSchedules");
    for (const cardId of affectedCardIds) {
      const history = await reviewLogs.index("by-card-id").getAll(cardId);
      if (history.length === 0) {
        await cardSchedules.delete(cardId);
      } else {
        await cardSchedules.put(fixedIntervalScheduler(history, new Date(importedAt)));
      }
    }

    await tx.done;
    return { importedAt, counts: countSavedData(data) };
  } catch (error) {
    try {
      tx.abort();
    } catch {
      // request failure で既に abort 済みなら InvalidStateError になるため再 abort は不要。
    }
    throw error;
  }
};
```

```ts
importSavedData(data) {
  const mergeById = <T extends { id: string }>(current: T[], incoming: T[]): T[] => [
    ...current.filter((item) => !incoming.some((candidate) => candidate.id === item.id)),
    ...incoming,
  ];
  const nextAnnotations = mergeById(annotations, data.annotations);
  const nextBookmarks = mergeById(bookmarks, data.bookmarks);
  const nextCollections = mergeById(collections, data.collections);
  const nextStudyCards = mergeById(studyCards, data.studyCards);
  const nextStudySessions = mergeById(studySessions, data.studySessions);
  const nextReviewLogs = mergeById(reviewLogs, data.reviewLogs);
  const affected = new Set([
    ...data.studyCards.map((card) => card.id),
    ...data.reviewLogs.map((log) => log.cardId),
  ]);
  for (const incomingLog of data.reviewLogs) {
    const previous = reviewLogs.find((log) => log.id === incomingLog.id);
    if (previous !== undefined) {
      affected.add(previous.cardId);
    }
  }
  const nextSchedules = cardSchedules.filter((schedule) => !affected.has(schedule.cardId));
  for (const cardId of affected) {
    const history = nextReviewLogs.filter((log) => log.cardId === cardId);
    if (history.length > 0) {
      nextSchedules.push(fixedIntervalScheduler(history, new Date(data.exportedAt)));
    }
  }

  annotations = nextAnnotations;
  bookmarks = nextBookmarks;
  collections = nextCollections;
  studyCards = nextStudyCards;
  studySessions = nextStudySessions;
  reviewLogs = nextReviewLogs;
  cardSchedules = nextSchedules;
  savedDocument = data.savedLaws[0] ?? savedDocument;
  savedAt = savedDocument?.savedAt;
  updatedAt = data.exportedAt;

  return Promise.resolve({ importedAt: data.exportedAt, counts: countSavedData(data) });
},
```

```ts
import type { SavedDataFile } from "@/core/native-integration";

export const downloadTextFile = (file: SavedDataFile): void => {
  const blob = new Blob([file.contents], { type: file.mediaType });
  const url = URL.createObjectURL(blob);
  const revokeObjectUrl = URL.revokeObjectURL.bind(URL);
  const link = document.createElement("a");

  try {
    link.href = url;
    link.download = file.fileName;
    document.body.append(link);
    link.click();
  } finally {
    link.remove();
    window.setTimeout(() => revokeObjectUrl(url), 100);
  }
};
```

```tsx
  applySavedDataImport,

import { Button } from "@/shared/ui/button";

import { downloadTextFile } from "./download-text-file";

const defaultStorageRepository = createStorageRepository();
type BusyState = "reading" | "exporting" | "importing";

const importErrorMessage = (error: unknown): string => {
  if (!(error instanceof SavedDataImportError)) {
    return "インポートできませんでした。データは変更されていません。再試行してください。";
  }
  if (error.code === "unsupported-version") {
    return "このファイルには対応していません。export version 2 のJSONを選択してください。";
  }
  if (error.code === "invalid-json") {
    return "JSONファイルとして読み取れませんでした。";
  }
  return `JSONの内容を検証できませんでした。${error.message}`;
};

export const DataTransferPage = ({
  storageRepository = defaultStorageRepository,
}: { storageRepository?: StorageRepository } = {}) => {
  const [busy, setBusy] = useState<BusyState>();
  const [prepared, setPrepared] = useState<PreparedSavedDataImport>();
  const [message, setMessage] = useState<string>();
  const [error, setError] = useState<string>();
  const disabled = busy !== undefined;

  const handleExport = async () => {
    if (disabled) return;
    setBusy("exporting");
    setMessage(undefined);
    setError(undefined);
    try {
      downloadTextFile(await createSavedDataFile(storageRepository, new Date()));
      setMessage("JSONを書き出しました。");
    } catch {
      setError("JSONを書き出せませんでした。保存データを確認して再試行してください。");
    } finally {
      setBusy(undefined);
    }
  };

  const handleFileChange = async (event: ChangeEvent<HTMLInputElement>) => {
    const file = event.currentTarget.files?.[0];
    event.currentTarget.value = "";
    if (file === undefined || disabled) return;
    setBusy("reading");
    setPrepared(undefined);
    setMessage(undefined);
    setError(undefined);
    let contents: string;
    try {
      contents = await file.text();
    } catch {
      setError("ファイルを読み込めませんでした。もう一度選択してください。");
      setBusy(undefined);
      return;
    }
    try {
      setPrepared(prepareSavedDataImportFile(contents));
    } catch (reason) {
      setError(importErrorMessage(reason));
    }
    setBusy(undefined);
  };

  const handleImport = async () => {
    if (prepared === undefined || disabled) return;
    setBusy("importing");
    setMessage(undefined);
    setError(undefined);
    try {
      const result = await applySavedDataImport(storageRepository, prepared);
      const total = Object.values(result.counts).reduce((sum, count) => sum + count, 0);
      setPrepared(undefined);
      setMessage(`7分類、合計${total.toLocaleString("ja-JP")}件のデータを取り込みました。`);
    } catch {
      setError("インポートできませんでした。データは変更されていません。再試行してください。");
    } finally {
      setBusy(undefined);
    }
  };

  const countRows =
    prepared === undefined
      ? []
      : ([
          ["保存法令本文", prepared.preview.counts.savedLaws],
          ["ブックマーク", prepared.preview.counts.bookmarks],
          ["コレクション", prepared.preview.counts.collections],
          ["メモ", prepared.preview.counts.annotations],
          ["学習カード", prepared.preview.counts.studyCards],
          ["回答ログ", prepared.preview.counts.reviewLogs],
          ["学習セッション", prepared.preview.counts.studySessions],
        ] as const);

  return (
    <section className="mx-auto grid w-full max-w-2xl gap-6 px-5 py-10">
      <div className="grid gap-2">
        <Link className="w-fit text-sm text-primary hover:underline" to="/settings">
          設定へ戻る
        </Link>
        <h1 className="font-serif text-2xl font-semibold">エクスポート / インポート</h1>
        <p className="text-sm leading-6 text-muted-foreground">
          JSONは端末内で処理され、外部へ送信されません。
        </p>
      </div>

      {message === undefined ? null : (
        <p role="status" className="rounded-md border p-3 text-sm">
          {message}
        </p>
      )}
      {error === undefined ? null : (
        <p
          role="alert"
          className="rounded-md border border-destructive p-3 text-sm text-destructive"
        >
          {error}
        </p>
      )}

      <section className="grid gap-3 rounded-md border bg-card p-4">
        <h2 className="font-semibold">JSONをエクスポート</h2>
        <p className="text-sm text-muted-foreground">
          保存法令本文と学習履歴を含むversion 2を書き出します。
        </p>
        <Button
          className="w-full sm:w-fit"
          disabled={disabled}
          onClick={() => void handleExport()}
          type="button"
          variant="outline"
        >
          <Download aria-hidden="true" />
          JSONをエクスポート
        </Button>
      </section>

      <section className="grid gap-3 rounded-md border bg-card p-4">
        <h2 className="font-semibold">JSONをインポート</h2>
        <label className="grid gap-2 text-sm font-medium">
          インポートするJSONファイル
          <Input
            accept=".json,application/json"
            disabled={disabled}
            onChange={(event) => void handleFileChange(event)}
            type="file"
          />
        </label>
        {busy === "reading" ? (
          <p role="status" className="text-sm text-muted-foreground">
            ファイルを確認しています。
          </p>
        ) : null}
        {prepared === undefined ? null : (
          <div className="grid gap-3 rounded-md border bg-background p-4">
            <div className="flex flex-wrap justify-between gap-2 text-sm">
              <span className="font-medium">version 2</span>
              <span className="text-muted-foreground">出力日時: {prepared.preview.exportedAt}</span>
            </div>
            <ul className="grid gap-1 text-sm">

            </Button>
          </div>
        )}
      </section>
    </section>
  );
};
```
