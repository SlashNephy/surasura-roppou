# 文字サイズ、行間、テーマ設定 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 文字サイズ、行間、テーマを設定画面から変更し、端末へ保存して主要表示面へ一貫して反映する。

**Architecture:** `core/settings` に検証付きの外部ストアを置き、React の Provider が `localStorage` と `next-themes` を DOM 属性へ橋渡しする。設定画面は Provider の hook だけを使い、CSS custom property と相対単位が法令本文、目次、検索結果、ダイアログを含む表示レイヤーを変更する。

**Tech Stack:** React 19、TypeScript 6、next-themes、Tailwind CSS 4、Vitest、Testing Library、playwright-cli

```ts
export const fontSizePreferences = ["standard", "large", "extra-large"] as const;
export type FontSizePreference = (typeof fontSizePreferences)[number];

export const lineSpacingPreferences = ["standard", "relaxed", "wide"] as const;
export type LineSpacingPreference = (typeof lineSpacingPreferences)[number];

export const themePreferences = ["system", "light", "dark"] as const;
export type ThemePreference = (typeof themePreferences)[number];

export const defaultFontSizePreference: FontSizePreference = "standard";
export const defaultLineSpacingPreference: LineSpacingPreference = "standard";
export const defaultThemePreference: ThemePreference = "system";

// 表示設定だけを削除、export、analytics の対象から分離できる名前空間に置く。
export const fontSizeStorageKey = "surasura:display:font-size";
export const lineSpacingStorageKey = "surasura:display:line-spacing";
export const themeStorageKey = "surasura:display:theme";

const listeners = new Set<() => void>();

const includes = <Value extends string>(values: readonly Value[], value: string): value is Value =>
  values.includes(value as Value);

export const isThemePreference = (value: string): value is ThemePreference =>
  includes(themePreferences, value);

export const isFontSizePreference = (value: string): value is FontSizePreference =>
  includes(fontSizePreferences, value);

export const isLineSpacingPreference = (value: string): value is LineSpacingPreference =>
  includes(lineSpacingPreferences, value);

const read = <Value extends string>(
  key: string,
  values: readonly Value[],
  fallback: Value,
): Value => {
  if (typeof localStorage === "undefined") {
    return fallback;
  }

  const stored = localStorage.getItem(key);
  return stored !== null && includes(values, stored) ? stored : fallback;
};

const write = (key: string, value: string): void => {
  if (typeof localStorage === "undefined") {
    return;
  }

  localStorage.setItem(key, value);
  for (const listener of listeners) {
    listener();
  }
};

export const getFontSizePreference = (): FontSizePreference =>
  read(fontSizeStorageKey, fontSizePreferences, defaultFontSizePreference);

export const getLineSpacingPreference = (): LineSpacingPreference =>
  read(lineSpacingStorageKey, lineSpacingPreferences, defaultLineSpacingPreference);

export const setFontSizePreference = (value: FontSizePreference): void => {
  write(fontSizeStorageKey, value);
};

export const setLineSpacingPreference = (value: LineSpacingPreference): void => {
  write(lineSpacingStorageKey, value);
};

export const sanitizeStoredThemePreference = (): ThemePreference => {
  if (typeof localStorage === "undefined") {
    return defaultThemePreference;
  }

  const stored = localStorage.getItem(themeStorageKey);
  if (stored === null) {
    return defaultThemePreference;
  }
  if (isThemePreference(stored)) {
    return stored;
  }

  localStorage.removeItem(themeStorageKey);
  return defaultThemePreference;
};

export const subscribeDisplayPreferences = (listener: () => void): (() => void) => {
  listeners.add(listener);

  const handleStorage = (event: StorageEvent) => {
    if (
      event.key === null ||
      event.key === fontSizeStorageKey ||
      event.key === lineSpacingStorageKey
    ) {
      listener();
    }
  };
  window.addEventListener("storage", handleStorage);

  return () => {
    listeners.delete(listener);
    window.removeEventListener("storage", handleStorage);
  };
};
```

const mediaListeners = new Set<(event: MediaQueryListEvent) => void>();

  ({
    media: query,
    get matches() {
      return prefersDark;

    onchange: null,
    addEventListener: (_type, listener) => mediaListeners.add(listener),
    removeEventListener: (_type, listener) => mediaListeners.delete(listener),

    removeListener: (listener) => mediaListeners.delete(listener),

```tsx
import { ThemeProvider, useTheme } from "next-themes";
import {
  type PropsWithChildren,
  useCallback,
  useLayoutEffect,
  useState,
  useSyncExternalStore,
} from "react";

import {
  defaultFontSizePreference,
  defaultLineSpacingPreference,
  defaultThemePreference,
  getFontSizePreference,
  getLineSpacingPreference,
  isThemePreference,
  sanitizeStoredThemePreference,
  setFontSizePreference,
  setLineSpacingPreference,
  subscribeDisplayPreferences,
  themeStorageKey,
  type FontSizePreference,
  type LineSpacingPreference,
  type ThemePreference,
} from "@/core/settings";

const DisplayPreferencesBridge = ({ children }: PropsWithChildren) => {
  const fontSize = useSyncExternalStore(
    subscribeDisplayPreferences,
    getFontSizePreference,
    () => defaultFontSizePreference,
  );
  const lineSpacing = useSyncExternalStore(
    subscribeDisplayPreferences,
    getLineSpacingPreference,
    () => defaultLineSpacingPreference,
  );

  useLayoutEffect(() => {
    document.documentElement.dataset.fontSize = fontSize;
    document.documentElement.dataset.lineSpacing = lineSpacing;
  }, [fontSize, lineSpacing]);

  return children;
};

export const DisplayPreferencesProvider = ({ children }: PropsWithChildren) => {
  const [defaultTheme] = useState(sanitizeStoredThemePreference);

  return (
    <ThemeProvider
      attribute="class"
      defaultTheme={defaultTheme}
      enableSystem
      storageKey={themeStorageKey}
    >
      <DisplayPreferencesBridge>{children}</DisplayPreferencesBridge>
    </ThemeProvider>
  );
};

export const useDisplayPreferences = () => {
  const fontSize = useSyncExternalStore(
    subscribeDisplayPreferences,
    getFontSizePreference,
    () => defaultFontSizePreference,
  );
  const lineSpacing = useSyncExternalStore(
    subscribeDisplayPreferences,
    getLineSpacingPreference,
    () => defaultLineSpacingPreference,
  );
  const { setTheme: setNextTheme, theme: nextTheme } = useTheme();
  const theme =
    nextTheme !== undefined && isThemePreference(nextTheme) ? nextTheme : defaultThemePreference;

  const setFontSize = useCallback((value: FontSizePreference) => setFontSizePreference(value), []);
  const setLineSpacing = useCallback(
    (value: LineSpacingPreference) => setLineSpacingPreference(value),
    [],
  );
  const setTheme = useCallback((value: ThemePreference) => setNextTheme(value), [setNextTheme]);

  return { fontSize, lineSpacing, setFontSize, setLineSpacing, setTheme, theme };
};
```
