# 本文ビューア MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** `/laws/:lawId` で代表的な法令本文を PC とスマートフォンの両方で読める本文ビューア MVP を実装する。

**Architecture:** 既存 `AppShell` の desktop 3 カラム / mobile bottom navigation を活かし、中央ペインに本文ビューア route を追加する。本文描画は `src/core/viewer/` に分離し、`LawNode[]` を domain model のまま受け取って node type ごとの見出し、条文 block、項・号インデントを表示する。データ取得は #14 では fixture-backed page adapter に留め、後続の repository / storage 接続に備えて discriminated union の page state で境界を作る。

**Tech Stack:** React 19, TanStack Router, TypeScript 6, Tailwind CSS 4, Vitest, Testing Library, lucide-react, shadcn/ui style primitives。

Create `src/core/viewer/LawNodeList.tsx`:

```tsx
import type { LawNode, LawNodeType } from "@/core/domain";
import { cn } from "@/shared/utils/cn";

interface LawNodeListProps {
  nodes: LawNode[];
}

const headingLevelByType = {
  Part: "text-xl font-semibold",
  Chapter: "text-lg font-semibold",
  Section: "text-base font-semibold",
  Subsection: "text-base font-semibold",
  Division: "text-base font-semibold",
  SupplementaryProvision: "text-lg font-semibold",
  AppdxTable: "text-lg font-semibold",
  AppdxStyle: "text-lg font-semibold",
} satisfies Partial<Record<LawNodeType, string>>;

export const LawNodeList = ({ nodes }: LawNodeListProps) => {
  const nodeById = new Map(nodes.map((node) => [node.id, node]));
  const topLevelNodes = nodes.filter((node) => node.parentId === undefined);

  return (
    <div className="grid gap-5">
      {topLevelNodes.map((node) => (
        <LawNodeBlock key={node.id} node={node} nodeById={nodeById} />
      ))}
    </div>
  );
};

const LawNodeBlock = ({ node, nodeById }: { node: LawNode; nodeById: Map<string, LawNode> }) => {
  const children = node.children
    .map((childId) => nodeById.get(childId))
    .filter((child): child is LawNode => child !== undefined);

  if (node.type === "Article") {
    return (
      <article
        aria-label={node.title ?? `条文 ${node.number ?? node.path}`}
        className="rounded-md border bg-background p-4 shadow-xs md:p-5"
      >
        <h2 className="text-lg font-semibold text-foreground">{node.title ?? node.number}</h2>
        <div className="mt-4 grid gap-3">
          {children.length === 0 ? (
            <p className="leading-8 text-foreground break-words">{node.plainText}</p>
          ) : (
            children.map((child) => (
              <LawNodeBlock key={child.id} node={child} nodeById={nodeById} />
            ))
          )}
        </div>
      </article>
    );
  }

  if (node.type === "Paragraph" || node.type === "Item" || node.type === "Subitem") {
    const childText = new Set(children.map((child) => child.plainText));
    const bodyText =
      children.length === 0 ? node.plainText : stripChildText(node.plainText, childText);

    return (
      <div
        className={cn(
          "grid gap-2",
          node.type === "Item" && "pl-5",
          node.type === "Subitem" && "pl-8",
        )}
      >
        <p className="flex min-w-0 gap-3 leading-8 text-foreground">
          {node.title !== undefined || node.number !== undefined ? (
            <span className="shrink-0 text-muted-foreground">{node.title ?? node.number}</span>
          ) : null}
          <span className="min-w-0 break-words">{bodyText}</span>
        </p>
        {children.map((child) => (
          <LawNodeBlock key={child.id} node={child} nodeById={nodeById} />
        ))}
      </div>
    );
  }

  const headingClassName = headingLevelByType[node.type];
  const title = node.title ?? node.plainText;

  return (
    <section className="grid gap-3">
      {headingClassName === undefined ? null : (
        <h2 className={cn("text-foreground break-words", headingClassName)}>{title}</h2>
      )}
      {children.length === 0 && headingClassName === undefined ? (
        <p className="leading-8 text-foreground break-words">{node.plainText}</p>
      ) : null}
      {children.map((child) => (
        <LawNodeBlock key={child.id} node={child} nodeById={nodeById} />
      ))}
    </section>
  );
};

const stripChildText = (plainText: string, childTexts: Set<string>): string => {
  let text = plainText;

  for (const childText of childTexts) {
    text = text.replace(childText, "").trim();
  }

  return text;
};
```

Create `src/app/law-viewer-page.tsx`:

```tsx
import { Link, useParams } from "@tanstack/react-router";
import { AlertTriangle, CloudOff } from "lucide-react";

import type { Law, LawNode, LawRevision } from "@/core/domain";
import { LawDocumentView } from "@/core/viewer";
import { Button } from "@/shared/ui/button";
import { Skeleton } from "@/shared/ui/skeleton";

import { sampleLawViewerDocument } from "./law-viewer-sample";

type LawViewerState =
  | { status: "loading" }
  | { status: "error"; message: string }
  | { status: "offline-unavailable"; lawTitle: string }
  | { status: "ready"; law: Law; revision: LawRevision; nodes: LawNode[]; isSaved: boolean };

export const LawViewerPage = () => {
  const { lawId } = useParams({ from: "/laws/$lawId" });

  return <LawViewerPageContent state={getLawViewerDocument(lawId)} />;
};

export const LawViewerPageContent = ({ state }: { state: LawViewerState }) => {
  if (state.status === "loading") {
    return (
      <section
        className="mx-auto grid w-full max-w-4xl gap-4 px-4 py-6 md:px-8"
        aria-label="読み込み中"
      >
        <Skeleton className="h-8 w-48" />
        <Skeleton className="h-24 w-full" />
        <Skeleton className="h-40 w-full" />
      </section>
    );
  }

  if (state.status === "error") {
    return (
      <section className="mx-auto grid min-h-[calc(100dvh-10rem)] w-full max-w-2xl place-content-center gap-4 px-4 py-10">
        <div role="alert" className="grid gap-3 rounded-md border bg-background p-5">
          <div className="flex items-center gap-2 text-destructive">
            <AlertTriangle className="size-5" aria-hidden="true" />
            <h1 className="text-lg font-semibold">本文を表示できません</h1>
          </div>
          <p className="leading-7 text-muted-foreground">{state.message}</p>
          <Button asChild variant="outline" className="w-fit">
            <Link to="/laws">法令検索へ戻る</Link>
          </Button>
        </div>
      </section>
    );
  }

  if (state.status === "offline-unavailable") {
    return (
      <section className="mx-auto grid min-h-[calc(100dvh-10rem)] w-full max-w-2xl place-content-center gap-4 px-4 py-10">
        <div role="status" className="grid gap-3 rounded-md border bg-background p-5">
          <div className="flex items-center gap-2 text-foreground">
            <CloudOff className="size-5 text-primary" aria-hidden="true" />
            <h1 className="text-lg font-semibold">{state.lawTitle}</h1>
          </div>
          <p className="leading-7 text-muted-foreground">この法令は端末に保存されていません。</p>
        </div>
      </section>
    );
  }

  return (
    <LawDocumentView
      law={state.law}
      revision={state.revision}
      nodes={state.nodes}
      isSaved={state.isSaved}
    />
  );
};

const getLawViewerDocument = (lawId: string): LawViewerState => {
  if (lawId === "offline-demo") {
    return { status: "offline-unavailable", lawTitle: "民法" };
  }

  if (lawId !== "129AC0000000089") {
    return { status: "error", message: "法令が見つかりません。" };
  }

  return { status: "ready", ...sampleLawViewerDocument };
};
```
