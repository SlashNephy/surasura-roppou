Create `references.ts` with:

```ts
export type ArticleReference = {
  lawId: string;
  revisionId?: string;
  article: string;
  paragraph?: string;
  item?: string;
};

export const buildArticleReferenceKey = (reference: ArticleReference): string => {
  const segments = [`law:${reference.lawId}`];

  if (reference.revisionId !== undefined) {
    segments.push(`revision:${reference.revisionId}`);
  }

  segments.push(`article:${reference.article}`);

  if (reference.paragraph !== undefined) {
    segments.push(`paragraph:${reference.paragraph}`);
  }

  if (reference.item !== undefined) {
    segments.push(`item:${reference.item}`);
  }

  return segments.map(encodeReferenceSegment).join("/");
};
```

Then add `buildLawArticleUrl`, `parseArticleReferenceKey`, and export from `index.ts`.
